      }

      }

      common /.style = {
        fontupper = \rmfamily,
        lower~separated = false,
        coltitle = white,
        colback = gray!5,
        boxrule = 0.5pt,
        fonttitle = \bfseries,
        enhanced,
        breakable,
        % top = 8pt,
        top = 12pt, % enlarge the top diatance to make up for the title box's height
        before~skip = 8pt,
        attach~boxed~title~to~top~left = {
          yshift = -0.125in,
          xshift = 0.15in
        },
        boxed~title~style = {
          boxrule = 0pt,
          colframe = white,
          arc = 0pt,
          outer~arc = 0pt
        },
        separator~sign={.}
      },
      defstyle /.style = {
        common,
        colframe = main,
        colback = main!5,
        colbacktitle = main,
        overlay~unbroken~and~last = {
          \node[anchor = south~east, outer~sep = 0pt] at (\linewidth-width, 0)
            { \textcolor{main}{$\clubsuit$} };
        }
      },
      thmstyle /.style = {
        common,
        colframe = second,
        colback = second!5,
        colbacktitle = second,
        overlay~unbroken~and~last = {
          \node[anchor = south~east, outer~sep = 0pt] at (\linewidth-width, 0)
            { \textcolor{second}{$\heartsuit$} };
        }
      },
      prostyle /.style = {
        common,
        colframe = third,
        colback = third!5,
        colbacktitle = third,
        overlay~unbroken~and~last = {
          \node[anchor = south~east, outer~sep = 0pt] at (\linewidth-width, 0)
            { \textcolor{third}{$\spadesuit$} };
        }
      },
      ELEGANT@title /.code~n~args = { 2 }
        {
          \tcbset
            {
              title =
                {
                  \cs:w #1name \cs_end:~
                  \ifdef{\thetcbcounter}{\thetcbcounter}{}
                  \tl_if_blank:nF { #2 } { \ (#2) }
                }
            }
        },
      ELEGANT@label /.code~n~args = { 2 }
        {
          \tl_if_blank:nF { #2 }
            { \tcbset { label = { #1:#2 } } }
        },
      ELEGANT@star@title /.code~n~args = { 2 }
        {
          \tcbset
            {
              title =
                {
                  \cs:w #1name \cs_end:
                  \tl_if_blank:nF { #2 } { \ (#2) }
                }
            }
        }
      }

    \cs_new_protected_nopar:Npn \__elegant_theorem_new_tcolorbox_environment:nnnnn #1#2#3#4#5

              }
          }

        \DeclareTColorBox { #1* } { g o }
          {

            IfValueTF = { ##1 }
              { ELEGANT@star@title = { #1 } { ##1 } }
              {
                IfValueTF = { ##2 }
                  { ELEGANT@star@title = { #1 } { ##2 } }
                  { ELEGANT@star@title = { #1 } { } }
              }
          }
      }

    \cs_new_protected_nopar:Npn \__elegant_theorem_declare_tcolorbox:nnnnn #1#2#3#4#5

      }

  }

\elegant_declare_theorem_mode:nn { simple }
	{
		\let\openbox\relax
		\RequirePackage{amsthm}
    
		\tl_new:N \l__elegant_tmp_cnt_name_tl 


		\cs_new_nopar:Npn \__elegant_theorem_new_enviroment_handle:nn #1#2 
			{
				\newtheoremstyle{#1}{3pt}{3pt}{\rmfamily}{-3pt}{
					\bfseries\color_select:o { \elegant_use_color:n { #1 } }}{}{0.5em}
					{
						\thmname{ ##1 }~\thmnumber{ \use:c { the#2} }~\thmnote{(##3) }
					}

				\theoremstyle{#1}
				\bool_if:NTF \g__elegant_usesamecnt_bool 
					{
						\newtheorem{#1}[elegant@theorem]{ \elegant_use_name:n { #1 } }
					}
					{
						\newtheorem{#1}{ \elegant_use_name:n { #1 } }[\g__elegant_thmcnt_tl]
					}
				\newtheorem*{#1*}{ \elegant_use_name:n { #1 } }
				\elegant_name_if_exist:nT { #1 / icon }
					{
						\AddToHook { env/#1/after }{ \hfill \textcolor{ \elegant_use_color:n { #1 } } { \elegant_use_name:n { #1 / icon } } }
						\AddToHook { env/#1*/after }{ \hfill \textcolor{ \elegant_use_color:n { #1 } } { \elegant_use_name:n { #1 / icon } } }
					}
			}
		\RenewDocumentCommand \elegantnewtheorem { m }
			{
				\bool_if:NTF \g__elegant_usesamecnt_bool 
          {
            \tl_set:Nn \l__elegant_tmp_cnt_name_tl { elegant@theorem }
          }
          {
            \tl_set:Nn \l__elegant_tmp_cnt_name_tl { #1 }
          }
					\exp_args:NnV \__elegant_theorem_new_enviroment_handle:nn { #1 } \l__elegant_tmp_cnt_name_tl
			}
		
		\clist_map_inline:nn 
      { theorem, definition, postulate, axiom, corollary, lemma, proposition }
      {
        \elegantnewtheorem { #1 }
      }
	}
