\begin{align}
a^2 + b^2 &= c^2 \label{eq:gougu} \\
\int_{0}^{1}\int_{0}^{1} f(x,y) dxdy &= 1 \label{eq:double} \\
\begin{vmatrix}
a & b \\
c & d
\end{vmatrix} &= ad-bc \label{eq:det}

\end{equation}
